028     * Aquest metode retorna el panell del taulell de joc, on es mostren les celles del joc.
029     * @param vistaPrincipal La vista principal de l'aplicacio.

031    public JugantView(PrincipalView vistaPrincipal) {
032        this.vistaPrincipal = vistaPrincipal;
033        initComponents();
034        running= false;
035        seconds = Duration.ofSeconds(0);
036        timer = new Timer(1000, new ActionListener(){
037            @Override
038            public void actionPerformed(ActionEvent e){
039                if(running){
040                    seconds = seconds.plusSeconds(1);
041                    long totalSeconds = seconds.getSeconds();
042                    long hours = totalSeconds / 3600;
043                    long minutes = (totalSeconds % 3600) / 60;
044                    long secs = totalSeconds % 60;
045                    tCrono1.setText(String.format("%02d:%02d:%02d", hours, minutes, secs));
046                }
047            }
048        });
049        timer.start();
050        
051        running = true;
052        
053        ImageIcon imageIcon = new javax.swing.ImageIcon(getClass().getResource("/images/flecha.png"));
054        Image image = imageIcon.getImage(); // Convertir ImageIcon a Image
055        Image newImage = image.getScaledInstance(50, 50, Image.SCALE_SMOOTH); // Escalar la imagen
056        ImageIcon scaledImageIcon = new ImageIcon(newImage);
057        bEnrere7.setIcon(scaledImageIcon);
058        
059        ImageIcon imageIcon2 = new javax.swing.ImageIcon(getClass().getResource("/images/goma.png"));
060        Image image2 = imageIcon2.getImage(); // Convertir ImageIcon a Image
061        Image newImage2 = image2.getScaledInstance(40, 40, Image.SCALE_SMOOTH); // Escalar la imagen
062        ImageIcon scaledImageIcon2 = new ImageIcon(newImage2);
063        bBorrar.setIcon(scaledImageIcon2);
064        
065        ImageIcon imageIcon3 = new javax.swing.ImageIcon(getClass().getResource("/images/clue.png"));
066        Image image3 = imageIcon3.getImage(); // Convertir ImageIcon a Image
067        Image newImage3 = image3.getScaledInstance(50, 50, Image.SCALE_SMOOTH); // Escalar la imagen
068        ImageIcon scaledImageIcon3 = new ImageIcon(newImage3);
069        bPista.setIcon(scaledImageIcon3);
070    }

273    private void bPistaActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_bPistaActionPerformed
274        // TODO add your handling code here:
275        //pedir pista
276        String t = tPista.getText();
277        Integer valor = Integer.parseInt(String.valueOf(t.charAt(8)));
278        if(valor >= 1){
279            Integer x = vistaPrincipal.getSeleccionado_x();
280            Integer y = vistaPrincipal.getSeleccionado_y();
281            if(x != -1 && y != -1){
282                try{
283                    vistaPrincipal.ajuda(x, y);
284                    valor = valor - 1;
285                    vistaPrincipal.getValorCela(x, y);
286                    //cosetes a fer
287                } catch (Exception e){
288                    if(e.getMessage().equals("Aquesta cella ja te un valor. Esborra'l abans de demanar pista.")){
289                        JOptionPane.showMessageDialog(null, "Error al solucionar el Kenken: " + e.getMessage(), "Warning", JOptionPane.WARNING_MESSAGE);
290                    } else {
291                        Object[] options = {"SI", "NO"};
292
293                        // Muestra el JOptionPane
294                        Integer option = JOptionPane.showOptionDialog(
295                            null,
296                            "La teva solucio proposada es incorrecta, vols saber una possible solucio d'aquesta cella?:",
297                            "Confirmacio",
298                            JOptionPane.YES_NO_OPTION,
299                            JOptionPane.QUESTION_MESSAGE,
300                            null,
301                            options,
302                            options[0]
303                        );
304
305
306
307                        // Verifica que opcion fue seleccionada
308                        if (option == JOptionPane.YES_OPTION) {
309                            try{
310                                vistaPrincipal.ajudaExtra(x, y);
311                                valor = valor - 1;
312                                vistaPrincipal.getValorCela(x, y);
313                            } catch (Exception e2){
314                                JOptionPane.showMessageDialog(null, "Eror al solucionar el Kenken: " + e2.getMessage(), "Warning", JOptionPane.WARNING_MESSAGE);
315                            }
316                        }
317                    }
318                }
319                tPista.setText("Pistas: "+String.valueOf(valor)+"/3");
320            } else JOptionPane.showMessageDialog(null, "Selecciona una cella Primer!", "Warning", JOptionPane.WARNING_MESSAGE);
321        }else{
322            JOptionPane.showMessageDialog(null, "No et queden Pistes", "Warning", JOptionPane.WARNING_MESSAGE);
323        }
324    }//GEN-LAST:event_bPistaActionPerformed
