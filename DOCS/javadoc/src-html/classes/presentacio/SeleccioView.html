048    public void crearpanels(JPanel cjt,String info,Boolean bd){
049        cjt.removeAll();
050        
051        
052        JsonArray jsonArray = JsonParser.parseString(info).getAsJsonArray();
053
054        cjt.setLayout(new FlowLayout(FlowLayout.LEFT));
055
056        
057        for (JsonElement element : jsonArray) {
058            
059            Boolean[] op = new Boolean[]{false,false,false,false,false,false,false};
060            JsonObject kenkenObject = element.getAsJsonObject();
061            javax.swing.JPanel kenken = new javax.swing.JPanel();
062            javax.swing.JLabel mida = new javax.swing.JLabel();
063            javax.swing.JPanel operacions = new javax.swing.JPanel();
064
065
066            kenken.setMaximumSize(new java.awt.Dimension(50,50));
067            kenken.setMinimumSize(new java.awt.Dimension(1, 1));
068            kenken.setPreferredSize(new java.awt.Dimension(50, 50));
069            kenken.setLayout(new java.awt.GridLayout(2, 1));
070            kenken.setBorder(javax.swing.BorderFactory.createBevelBorder(javax.swing.border.BevelBorder.RAISED));
071
072            mida.setFont(new java.awt.Font("Segoe UI", 1, 12)); // NOI18N
073            mida.setHorizontalAlignment(javax.swing.SwingConstants.CENTER);
074            
075            int N = kenkenObject.has("N") ? kenkenObject.get("N").getAsInt() : 0;
076
077            
078            JsonArray conjuntsCelesArray = kenkenObject.getAsJsonArray("regions");
079            for (Integer i = 0; i < conjuntsCelesArray.size(); ++i) {
080
081                JsonObject ConjuntCellaObj = conjuntsCelesArray.get(i).getAsJsonObject();
082                int index; 
083                index = ConjuntCellaObj.has("oper") ? ConjuntCellaObj.get("oper").getAsInt() : 0;
084                op[index] = true;
085                //cortarlo cuando todos sean true
086                
087                
088            }
089            op[0] = false;
090            mida.setText(N+"x"+N);
091            mida.setAlignmentX(0.5F);
092            mida.setHorizontalTextPosition(javax.swing.SwingConstants.CENTER);
093            kenken.add(mida);
094            
095            kenken.putClientProperty("id", kenkenObject.has("id") ? kenkenObject.get("id").getAsInt() : -1);
096            kenken.putClientProperty("bd", bd);
097            
098            operacions.setLayout(new java.awt.GridLayout());
099            for(int i= 0; i <= 6; ++i){
100                if(op[i]){
101                    JLabel la_op = new JLabel();
102                    la_op.setText(vistaPrincipal.calcularSimbol(i));
103                    la_op.setHorizontalAlignment(javax.swing.SwingConstants.CENTER);
104                    la_op.setBorder(javax.swing.BorderFactory.createEtchedBorder());
105                    operacions.add(la_op);
106                }
107            }
108            kenken.add(operacions);
109            kenken.setCursor(Cursor.getPredefinedCursor(Cursor.HAND_CURSOR));
110            
111            kenken.addMouseListener(new MouseAdapter() {
112                @Override
113                public void mouseClicked(MouseEvent e) {
114                    int id = (Integer) kenken.getClientProperty("id");
115                    Boolean bd = (Boolean) kenken.getClientProperty("bd");
116                    vistaPrincipal.setIdKenken(id);
117                    vistaPrincipal.setBd(bd);
118                    vistaPrincipal.showPanel("Seleccio2View");
119                  
120                }
121            });
122            
123            cjt.add(kenken);
124            
125
126            
127        }
128        
129
130        
131    }
