367    private void bGenerarKenken1ActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_bGenerarKenken1ActionPerformed
368
369        Integer mida, nColocades, maxCjt;
370
371        Boolean[] operacions = new Boolean[7];
372        operacions[0] = true;
373        operacions[1] = cSuma.isSelected();
374        operacions[2] = cResta.isSelected();
375        operacions[3] = cMultiplicacio.isSelected();
376        operacions[4] = cDivisio.isSelected();
377        operacions[5] = cMax.isSelected();
378        operacions[6] = cMin.isSelected();
379
380        if(!isNumber(aMida.getText()) && !aMida.getText().equals(text_mida) ){
381            JOptionPane.showMessageDialog(null, "El valor de mida ha de estar entre 1 i 12", "Advertencia", JOptionPane.WARNING_MESSAGE);
382            return;
383        }
384
385        if(!isNumber(aColocades.getText()) && !aColocades.getText().equals(text_colocades) ){
386            JOptionPane.showMessageDialog(null, "El numero de colocades ha de ser inferior a la mida i no pot ser negatiu", "Advertencia", JOptionPane.WARNING_MESSAGE);
387            return;
388        }
389
390        if(!isNumber(aMax.getText()) && !aMax.getText().equals(text_max) ){
391            JOptionPane.showMessageDialog(null, "El numero max de celes x cjt ha de ser inferior a 6 i no pot ser negatiu", "Advertencia", JOptionPane.WARNING_MESSAGE);
392            return;
393        }
394
395        if(aMida.getText().equals(text_mida) || Integer.valueOf(aMida.getText()) < 1 || Integer.valueOf(aMida.getText()) > 12 ){
396
397            JOptionPane.showMessageDialog(null, "El valor de mida ha de estar entre 1 i 12", "Advertencia", JOptionPane.WARNING_MESSAGE);
398            mida = 0;
399            return;
400
401        }else{
402            System.out.println(isNumber(aMida.getText()));
403            mida = Integer.valueOf(aMida.getText());
404        }
405
406        if(aColocades.getText().equals(text_colocades)){
407            nColocades = 0;
408        }else{
409            if(Integer.valueOf(aColocades.getText()) > mida || Integer.valueOf(aColocades.getText()) < 0 ||!isNumber(aColocades.getText())){
410                JOptionPane.showMessageDialog(null, "El numero de colocades ha de ser inferior a la mida i no pot ser negatiu", "Advertencia", JOptionPane.WARNING_MESSAGE);
411                return;
412            }else{
413                nColocades = Integer.valueOf(aColocades.getText());
414            }
415        }
416
417        if(aMax.getText().equals(text_max)){
418            maxCjt = 3;
419        }else{
420            if(Integer.valueOf(aMax.getText()) > 5 || Integer.valueOf(aMax.getText()) < 0 || !isNumber(aMax.getText())){
421                JOptionPane.showMessageDialog(null, "El numero max de celes x cjt ha de ser inferior a 6 i no pot ser negatiu", "Advertencia", JOptionPane.WARNING_MESSAGE);
422                return;
423            }else{
424                maxCjt = Integer.valueOf(aMax.getText());
425            }
426        }
427        
428        if(operacions[1] == false && operacions[3] == false && operacions[5] == false && operacions[6] == false ){
429
430            JOptionPane.showMessageDialog(null, "Has de seleccioanr minim una operacio basica", "Advertencia", JOptionPane.WARNING_MESSAGE);
431            return;
432
433        }
434        
435
436        try {
437            Integer idKenken = vistaPrincipal.generarKenken(mida, maxCjt, nColocades, operacions);
438            vistaPrincipal.setBd(Boolean.FALSE);
439            vistaPrincipal.setIdKenken(idKenken);
440            
441            //vistaPrincipal.crearKenken2(kenkenJSON,"JugantView","jugable");
442            //vistaPrincipal.showPanel("JugantView");
443            String nom = vistaPrincipal.getNomJugador();
444            vistaPrincipal.crearPartida(idKenken, nom);
445            String kenkenJSON = vistaPrincipal.obtenirKenkenPartida(idKenken,nom);
446            System.out.println("kenkenJSON: "+kenkenJSON);
447            vistaPrincipal.crearKenken2(kenkenJSON,"JugantView","jugable");
448            vistaPrincipal.showPanel("JugantView");
449        } catch (Exception e) {
450            // Handle the exception here
451            JOptionPane.showMessageDialog(null, "Error en generar el Kenken" + e.getMessage(), "Error", JOptionPane.ERROR_MESSAGE);
452            //e.printStackTrace();
453        }
454
455    }//GEN-LAST:event_bGenerarKenken1ActionPerformed
