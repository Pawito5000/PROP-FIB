197    private void bIniciarSessio2ActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_bIniciarSessio2ActionPerformed
198        // TODO add your handling code here:
199        String nom = aNom1.getText();
200        String contra = aContrasenya1.getText();
201        if(nom.equals("Introdueix el teu nom d'usuari") || nom.contains(" ")) {
202            JOptionPane.showMessageDialog(null, "Introdueix al teu nom d'usuari sense espais", "Warning", JOptionPane.WARNING_MESSAGE);
203        } else if (contra.equals("aaaaaaaaaaaaa") || String.valueOf(aContrasenya1.getPassword()).isEmpty() || contra.contains(" ")){
204            JOptionPane.showMessageDialog(null, "Introdueix la teva contrasenya sense espais", "Warning", JOptionPane.WARNING_MESSAGE);
205        } else {
206            if(vistaPrincipal.existsUsuari(nom)){
207                try {
208                    vistaPrincipal.getUsuari(nom);
209                    vistaPrincipal.verificarContra(contra);
210                    vistaPrincipal.setUsuari(nom);
211                    vistaPrincipal.showPanel("JocView");
212                } catch (Exception e){ 
213                    JOptionPane.showMessageDialog(null, "Error al recuperar Usuari: " + e.getMessage(), "Error", JOptionPane.ERROR_MESSAGE);
214                }
215            } else {
216                JOptionPane.showMessageDialog(null, "L'usuari no existeix al sistema", "Error", JOptionPane.ERROR_MESSAGE);
217            }
218            
219        }
220        
221    }//GEN-LAST:event_bIniciarSessio2ActionPerformed
