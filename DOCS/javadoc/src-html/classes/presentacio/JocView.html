176    private ArrayList<String> readFile(File file) {
177        ArrayList<String> lines = new ArrayList<>() ;
178        try (BufferedReader reader = new BufferedReader(new FileReader(file))) {
179            String line;
180            Integer i= 0;
181            while ((line = reader.readLine()) != null) {
182                lines.add(line);
183                ++i;
184            }
185        } catch (IOException e) {
186            JOptionPane.showMessageDialog(this, "Error reading file: " + e.getMessage(), "Error", JOptionPane.ERROR_MESSAGE);
187        }
188        return lines;
189    }
