049    public void ini(String Ranking,String nomJugador){
050        if(vistaPrincipal.getIdPartida() == -1){ //ve de la pantalla de seleccio
051            tPantalla6.setText("Ranking");
052            bContinuar1.setVisible(false);
053            bEnrere7.setVisible(true);
054        }
055        else{//acaba una partida
056            tPantalla6.setText("Enhorabona!!!");
057            bContinuar1.setVisible(true);
058            bEnrere7.setVisible(false);
059        }
060        
061        JsonObject RankingJSON = new Gson().fromJson(Ranking,JsonObject.class);
062        
063        JsonArray posicions = RankingJSON.getAsJsonArray("rankings");
064        
065        Boolean trobat = false;
066        
067        Integer mida = posicions.size();
068        if(mida > 10) mida = 10;
069        
070        for (Integer i = 0; i < mida; ++i) {
071            posiciones[i].setFont(new java.awt.Font("Segoe UI Semibold", 0, 18));
072            tiempos[i].setFont(new java.awt.Font("Segoe UI", 0, 14));
073            nombres[i].setFont(new java.awt.Font("Segoe UI", 0, 14));
074            JsonObject posicio = posicions.get(i).getAsJsonObject();
075            String pos = String.valueOf(i+1);
076            String temps = posicio.has("tiempo") ? posicio.get("tiempo").getAsString() : "NULL";
077            String nom = posicio.has("username") ? posicio.get("username").getAsString() : "NULL";
078
079            
080            posiciones[i].setText(pos);
081            tiempos[i].setText(temps);
082            nombres[i].setText(nom);
083            if(nom.equals( nomJugador)){
084                trobat = true;
085                posiciones[i].setFont(new java.awt.Font("Segoe UI Semibold", 1, 18));
086                tiempos[i].setFont(new java.awt.Font("Segoe UI", 1, 14));
087                nombres[i].setFont(new java.awt.Font("Segoe UI", 1, 14));
088            }
089            
090        }
091        
092        if(trobat == false){
093            String[] info= vistaPrincipal.getInfoRanking(vistaPrincipal.getIdKenken(),nomJugador);
094            if(info != null){
095                posiciones[8].setText("...");
096                tiempos[8].setText("...");
097                nombres[8].setText("...");
098                
099                posiciones[9].setFont(new java.awt.Font("Segoe UI Semibold", 1, 18));
100                tiempos[9].setFont(new java.awt.Font("Segoe UI", 1, 14));
101                nombres[9].setFont(new java.awt.Font("Segoe UI", 1, 14));
102                
103                posiciones[9].setText(info[0]);
104                tiempos[9].setText(info[1]);
105                nombres[9].setText(nomJugador);
106            }   
107
108        }
109        
110        
111    }
