003import java.util.AbstractMap;

009/**
010 * Representa una cella del Kenken.
011 * 
012 * Una cella pot tenir un valor inicial, un valor, una solucio i possibles solucions.
013 * Tambe proporciona metodes per establir i obtenir els valors de la cella.
014 */
015public class Cella {
016    private AbstractMap.SimpleEntry<Integer, Integer> coord;
017    private boolean inicial;
018    private Integer value = 0;
019    private Integer solucio = 0;
020     
021
022    private  HashMap<Integer, AbstractMap.SimpleEntry<Integer, Boolean>> idSoltoValor; //id valor bool
023
024    
025    
026    /**
027     * Constructor de la classe Cella.
028     * 
029     * @param PosX La coordenada x de la cella. Ha de ser un enter mes gran o igual a 0.
030     * @param PosY La coordenada y de la cella. Ha de ser un enter mes gran o igual a 0.
031     * @param value El valor inicial de la cella. Si es null, el valor de la cella es posa a 0.
032     */
033    public Cella(int PosX, int PosY, Integer value) 
034    {
035        this.coord = new AbstractMap.SimpleEntry<>(PosX, PosY);
036        if (value == null) this.value = 0;
037        else this.value = value;
038        this.inicial = false;
039        idSoltoValor = new HashMap<Integer, AbstractMap.SimpleEntry<Integer, Boolean>>();
040    }
041
042    /**
043     * Constructor de la classe Cella.
044     * 
045     * @param PosX La coordenada x de la cella. Ha de ser un enter mes gran o igual a 0.
046     * @param PosY La coordenada y de la cella. Ha de ser un enter mes gran o igual a 0.
047     * @param value El valor inicial de la cella. Si es null, el valor de la cella es posa a 0.
048     * @param inicial Indica si la cella es inicial o no.
049     */
050    public Cella(int PosX, int PosY, Integer value, boolean inicial) 
051    {
052        this.coord = new AbstractMap.SimpleEntry<>(PosX, PosY);
053        if (value == null) this.value = 0;
054        else this.value = value;
055        this.inicial = inicial;
056        if(inicial) this.solucio = value;
057        idSoltoValor = new HashMap<Integer, AbstractMap.SimpleEntry<Integer, Boolean>>();
058
059    }
060
061    /**
062     * @return JsonObject es genera un Json del objecte Cella
063     */
064    public JsonObject generateJSON() {
065        JsonObject kenkenObject = new JsonObject();
066        kenkenObject.addProperty("x", coord.getKey());
067        kenkenObject.addProperty("y", coord.getValue());
068        if(value != null){
069            kenkenObject.addProperty("cont", value);
070            kenkenObject.addProperty("ini", inicial ? 1 : 0);
071        } 
072        return kenkenObject;
073    }
074    public JsonObject generateSolucio() {
075        JsonObject kenkenObject = new JsonObject();
076        kenkenObject.addProperty("x", coord.getKey());
077        kenkenObject.addProperty("y", coord.getValue());
078        if(value != null){
079            kenkenObject.addProperty("cont", solucio);
080            kenkenObject.addProperty("ini", 1);
081        } 
082        return kenkenObject;
083    }

198
199    /**
200     * Colloca la solucio de la cella.
201     * 
202     * @param id_solucio L'identificador de la solucio.
203     */
204    public void colocarSolucio(Integer id_solucio)
205    {
206        if (!this.inicial && idSoltoValor.containsKey(id_solucio)){
207            this.solucio = idSoltoValor.get(id_solucio).getKey();
208        }
209    }
210
211    /**
212     * Afegeix el valor d'una possible solucio de la cella.
213     * 
214     * @param id_solucio L'identificador de la solucio.
215     * @param valor El valor a comprovar.
216     */
217    public void afegirPossibilitatTrue(Integer id_solucio, Integer valor)
218    {
219        AbstractMap.SimpleEntry<Integer, Boolean> valor_bool = new AbstractMap.SimpleEntry<Integer, Boolean>(valor, true);
220        idSoltoValor.put(id_solucio, valor_bool);
221    }
222
223    /**
224     * Elimina una possible solucio de la cella.
225     * 
226     * @param valor El valor a eliminar.
227     * @return {@code Set<Integer>}
228     */
229    public Set<Integer> eliminaSolucio(Integer valor)
230    {
231        Set<Integer> ids_eliminats = new HashSet<>();
232
233        for (Integer id : idSoltoValor.keySet()) {
234            if (idSoltoValor.get(id).getKey()==valor && idSoltoValor.get(id).getValue() ) {//conte el id
235                ids_eliminats.add(id);
236                AbstractMap.SimpleEntry<Integer, Boolean> valor2 = new AbstractMap.SimpleEntry<>(valor, false);
237                idSoltoValor.put(id, valor2);
238            }
239        }
240
241        return ids_eliminats;
242    }
243
244    /**
245     * Estableix una possible solucio de la cella a false.
246     * 
247     * @param id_solucio L'identificador de la solucio.
248     */
249    public void setPossibilitatFalse(Integer id_solucio){
250        if(idSoltoValor.containsKey(id_solucio)){
251            idSoltoValor.get(id_solucio).setValue(false);
252        }
253    }
254
255
256    /**
257     * Desfer l'eliminacio d'una possible solucio de la cella.
258     */
259    public void desferEliminarSolucio(Integer id_solucio){
260        if(idSoltoValor.containsKey(id_solucio)){
261            idSoltoValor.get(id_solucio).setValue(true);
262        }
263    }
264
265    /**
266     * Retorna les possibles solucions de la cella.
267     * 
268     * @return {@code Set<Integer>}
269     */
270    public Set<Integer> getPosiblitats(){
271        Set<Integer> posiblitats = new HashSet<>();
272        for (Integer id : idSoltoValor.keySet()) {
273            if (idSoltoValor.get(id).getValue()) posiblitats.add(id); //nomes les qeu siguin true
274        }
275        return posiblitats;
276    }
277
278    /**
279     * Retorna les possibles solucions de la cella.
280     * 
281     * @return {@code HashMap<Integer, AbstractMap.SimpleEntry<Integer, Boolean>>}
282     */
283    public HashMap<Integer, AbstractMap.SimpleEntry<Integer, Boolean>> getIdSoltoValor(){
284        return idSoltoValor;
285    }
