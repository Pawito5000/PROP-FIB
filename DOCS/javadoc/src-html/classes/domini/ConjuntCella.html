037    /**
038     * Creadora de la clase ConjuntCella a partir dels paramertres del Json
039     * 
040     * @param ConjuntCellaObj L'objeto Json que representa el conjunt de celles.
041     * @param kenken L'objeto Kenken al qual pertany el conjunt.
042     * 
043     * @throws IllegalArgumentException Si el nombre de celles no coincideix amb el nombre de celles del conjunt.
044     */
045    public ConjuntCella(JsonObject ConjuntCellaObj, Kenken kenken) 
046    {
047        this.resultat = 0;
048        this.ncColocades = 0;
049        this.coordToCella = new LinkedHashMap<>();
050        this.kenken = kenken;
051        this.numPossibilitats = null;
052
053        if (ConjuntCellaObj != null) {
054            this.nOperacio = ConjuntCellaObj.has("oper") ? ConjuntCellaObj.get("oper").getAsInt() : 0;
055            this.resultat = ConjuntCellaObj.has("result") ? ConjuntCellaObj.get("result").getAsInt() : 0;
056            int numcaselles = ConjuntCellaObj.has("e") ? ConjuntCellaObj.get("e").getAsInt() : 0;
057
058        this.nOperacio = ConjuntCellaObj.get("oper").getAsInt();
059        this.resultat = ConjuntCellaObj.get("result").getAsInt();
060        Integer celes = ConjuntCellaObj.get("e").getAsInt();
061
062            operacio = FactoriaOperacions.crearOperacio(nOperacio);
063
064            if (ConjuntCellaObj.has("coordenades")) {
065                JsonArray CelesArray = ConjuntCellaObj.getAsJsonArray("coordenades");
066                if(numcaselles != CelesArray.size()) throw new IllegalArgumentException("El nombre de celles no coincideix amb el nombre de celles del conjunt");
067                Integer minCoordenada_x = -1;
068                Integer minCoordenada_y = -1;
069                Integer maxCoordenada_x = -1;
070                Integer maxCoordenada_y = -1;
071                AbstractMap.SimpleEntry<Integer, Integer> coord_aux = null;
072                for (Integer i = 0; i < numcaselles; ++i) {
073                    JsonObject celaObject = CelesArray.get(i).getAsJsonObject();
074                    Integer x = celaObject.has("x") ? celaObject.get("x").getAsInt() : 0;
075                    if(x == 0) throw new IllegalArgumentException("La coordenada x no pot ser 0");
076                    Integer y = celaObject.has("y") ? celaObject.get("y").getAsInt() : 0;
077                    if(y == 0) throw new IllegalArgumentException("La coordenada y no pot ser 0");
078                    if(minCoordenada_x == -1 || x < minCoordenada_x) minCoordenada_x = x;
079                    if(minCoordenada_y == -1 || y < minCoordenada_y) minCoordenada_y = y;
080                    if(maxCoordenada_x == -1 || x > maxCoordenada_x) maxCoordenada_x = x;
081                    if(maxCoordenada_y == -1 || y > maxCoordenada_y) maxCoordenada_y = y;
082                    if(coord_aux == null) coord_aux = new AbstractMap.SimpleEntry<>(x, y);
083                    else if(coord_aux.equals(new AbstractMap.SimpleEntry<>(x, y))) throw new IllegalArgumentException("Hi ha celles repetides");
084                    Integer cont = celaObject.has("cont") ? celaObject.get("cont").getAsInt() : 0;
085                    Integer valorEnt_aux = celaObject.has("ini") ? celaObject.get("ini").getAsInt() : 0;
086                    Boolean ini = (valorEnt_aux != 0);
087                    Cella c = new Cella(x, y, cont, ini);
088                    coordToCella.put(new AbstractMap.SimpleEntry<>(x, y), c);
089                    if (cont != 0) ncColocades++;
090                }
091                Set<AbstractMap.SimpleEntry<Integer, Integer>> visited = new HashSet<>();
092                dfs(coord_aux.getKey(), coord_aux.getValue(), minCoordenada_x, minCoordenada_y, maxCoordenada_x, maxCoordenada_y, visited);
093                // Verificar si todas las celdas de la region han sido visitadas
094                for (AbstractMap.SimpleEntry<Integer, Integer> cell : coordToCella.keySet()) {
095                    if (!visited.contains(cell)) {
096                        throw new IllegalArgumentException("No totes les celles de la regio estan connectades" + cell.getKey() + " " + cell.getValue());
097                    }
098                }
099            }
100        }
101    }

104    /** 
105     * Recorre tot el conjunt de celles de la regio de manera recursiva per comprovar que totes les celles estan connectades
106
107     * @param visited  set de celles visitades
108     * @param x coordenada x
109     * @param y coordenada y
110     * @param minX limit inferior de x
111     * @param minY limit inferior de y
112     * @param maxX limit superior de x
113     * @param maxY limit superior de y
114     */
115    private void dfs(int x, int y, int minX, int minY, int maxX, int maxY, Set<AbstractMap.SimpleEntry<Integer, Integer>> visited) {
116        AbstractMap.SimpleEntry<Integer, Integer> cell = new AbstractMap.SimpleEntry<>(x, y);
117        if (x < minX || x > maxX || y < minY || y > maxY || visited.contains(cell) || !coordToCella.containsKey(cell)) {
118            return; // Fuera de los limites o celda ya visitada o no pertenece a la region
119        }
120        visited.add(cell); // Marcar la celda como visitada
121        // Explorar las celdas adyacentes
122        dfs(x + 1, y, minX, minY, maxX, maxY, visited); // Derecha
123        dfs(x - 1, y, minX, minY, maxX, maxY, visited); // Izquierda
124        dfs(x, y + 1, minX, minY, maxX, maxY, visited); // Abajo
125        dfs(x, y - 1, minX, minY, maxX, maxY, visited); // Arriba
126    }

133        JsonObject kenkenObject = new JsonObject();
134
135        kenkenObject.addProperty("oper", nOperacio);
136        kenkenObject.addProperty("result", resultat);
137        kenkenObject.addProperty("e", coordToCella.size());
138
139
140        JsonArray CelesArray = new JsonArray();
141        for (Cella cela : coordToCella.values()) {
142            JsonObject CellaObj = cela.generateJSON();
143            CelesArray.add(CellaObj);
144        }
145        kenkenObject.add("coordenades", CelesArray);
146
147        return kenkenObject;
148    }

433     * Coloca la solucio amb id_solucio a cada una de les celes del conjunt
434     *
435     * @param id_cjt l'identificador de un conjunt
436     * @param id_solucio l'identificador de la solucio
437     * @return els id_solucio eliminats al colocar la solucio.
438     */
439    public HashSet<Integer> colocarSolucio(Integer id_cjt, Integer id_solucio){
440        
441        HashSet<Integer> ids_fila = new HashSet<>();
442        HashSet<Integer> ids_columna = new HashSet<>();
443        HashSet<Integer> ids_totals = new HashSet<>();
444        
445        for (AbstractMap.SimpleEntry<Integer, Integer> coord : coordToCella.keySet()) {
446            Cella cella = coordToCella.get(coord);
447            cella.colocarSolucio(id_solucio);
448
449            Integer sol = cella.getSolucio();
450            ids_fila = kenken.descartarSolucionsFila(coord.getKey(),sol, id_cjt); 
451            ids_columna = kenken.descartarSolucionsColumna(coord.getValue(),sol, id_cjt);
452            ids_totals.addAll(ids_columna);
453            ids_totals.addAll(ids_fila);
454
455        }
456
457        return ids_totals;
458    }
