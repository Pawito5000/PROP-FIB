001package classes.domini;
002
003import com.google.gson.Gson;
004import com.google.gson.JsonObject;
005
006/**
007 * Representa un usuari del sistema.
008 * 
009 * Un usuari te un identificador, un nom i una contrasenya.
010 * Tambe proporciona metodes per obtenir i comprovar la contrasenya de l'usuari.
011 */
012public class Usuari {
013    private static Usuari instance;
014    private int id;
015    private String nom;
016    private String contrasenya;
017    private static int contador = 0;
018    private Integer contKenken;
019
020    /**
021     * Constructor de la classe Usuari.
022     * 
023     * @param nom Nom de l'usuari.
024     * @param contrasenya Contrasenya de l'usuari.
025     */
026    public Usuari(String nom, String contrasenya) {
027      
028        this.id = contador;
029        ++contador;
030        this.nom = nom;
031        this.contrasenya = contrasenya;
032        this.contKenken = 0;
033    }
034
035    /**
036     * Constructor de la classe Usuari a partir d'un JsonObject.
037     * 
038     * @param UsuariJSON String en format JSON amb la informacio de l'usuari.
039     */
040    public Usuari(String UsuariJSON) {
041        JsonObject UsuariObject = new Gson().fromJson(UsuariJSON, JsonObject.class);
042        this.id = UsuariObject.get("id").getAsInt();
043        this.nom = UsuariObject.get("nom").getAsString();
044        this.contrasenya = UsuariObject.get("contrasenya").getAsString();
045        this.contKenken = UsuariObject.get("contador").getAsInt();
046    }
047
048    /**
049     * @return JsonObject es genera un Json del objecte Usuari
050     */
051    public JsonObject toJson() {
052        JsonObject UsuariJSON = new JsonObject();
053        UsuariJSON.addProperty("id", id);
054        UsuariJSON.addProperty("nom", nom);
055        UsuariJSON.addProperty("contrasenya", contrasenya);
056        UsuariJSON.addProperty("contador", contKenken);
057        return UsuariJSON;
058    }
059
060    /**
061     * Retorna la instancia de l'usuari.
062     * 
063     * @param nom Nom de l'usuari.
064     * @param contrasenya Contrasenya de l'usuari.
065     * @return Usuari
066     */
067    public static Usuari getInstancia(String nom, String contrasenya) {
068        if (instance == null) {
069            instance = new Usuari(nom, contrasenya);
070        }
071        return instance;
072    }
073
074    /**
075     * Retorna l'indentificador de l'usuari.
076     * 
077     * @return Integer
078     */
079    public int getId() {
080        return id;
081    }
082
083    /**
084     * Retorna el nom de l'usuari.
085     * 
086     * @return String
087     */
088    public String getNom() {
089        return nom;
090    }
091
092    /**
093     * Incrementa el contador de Kenken.
094     */
095    public void incrementarContador() {
096        ++contKenken;
097    }
098
099    /**
100     * Obtenim el contador de Kenken.
101     * 
102     * @return
103     */
104    public Integer getContadorKenken() {
105        return contKenken;
106    }
107
108    /**
109     * Retorna la contrasenya de l'usuari.
110     * 
111     * @param contrasenya Contrasenya de l'usuari.
112     * @return boolean
113     */
114    public boolean comprobarContrasenya(String contrasenya) {
115        return this.contrasenya.equals(contrasenya);
116    }
117}
