039    /**
040     * Construeix un nou objecte Kenken a partir de l'objecte JSON proporcionat.
041     *
042     * @param KenkenJSONString String JSON object que representa un Kenken
043     */
044    public Kenken(String KenkenJSONString) {
045        JsonObject KenkenJSON = new Gson().fromJson(KenkenJSONString, JsonObject.class);
046        ctrl_domini = Ctrl_Domini.getInstancia();
047        if(KenkenJSON.has("id")) this.id = KenkenJSON.get("id").getAsInt();
048        else {
049            this.id = ctrl_domini.getContadorKenken();
050            ctrl_domini.incrementarContador();
051        }
052        this.cjtCeles = new HashMap<>();
053        this.coordToNumCjt = new HashMap<>();
054
055        this.mida = KenkenJSON.has("N") ? KenkenJSON.get("N").getAsInt() : 0;
056        Integer numCjtCela = KenkenJSON.has("R") ? KenkenJSON.get("R").getAsInt() : 0;
057        this.ncColocades = 0;
058
059        JsonArray conjuntsCelesArray = KenkenJSON.getAsJsonArray("regions");
060        for (Integer i = 0; i < conjuntsCelesArray.size(); ++i) {
061            JsonObject ConjuntCellaObj = conjuntsCelesArray.get(i).getAsJsonObject();
062            ConjuntCella cjt = new ConjuntCella(ConjuntCellaObj, this);
063            this.ncColocades += cjt.getNcColocades();
064            
065
066            // insert en el hashmap cjtCeles amb int i and CjtCela
067            cjtCeles.put(i, cjt);
068            // agafar les coordenades del cjt
069            List<AbstractMap.SimpleEntry<Integer, Integer>> coords = cjt.getCoordenades();
070            // afegir totes les coordenades de la llista al hashmap cjtCeles amb les coords i el int i
071            for (AbstractMap.SimpleEntry<Integer, Integer> coord : coords) {
072                coordToNumCjt.put(coord, i);
073            }
074        }
075    }

180    /**
181     * Comprova si es repeteixen valor en una fila en l'atribut valor
182     * 
183     * @param x Fila de la cela a comprovar
184     * @param valor Valor a comprovar
185     * @return Retorna true si existeix1 i tan sols 1 cela amb aquest valor en la fila
186     *
187     */
188    public boolean comprovarFila(Integer x, Integer valor) {
189        if (x > mida || x <= 0)
190            throw new IllegalArgumentException("La fila demanada esta fora dels limits");
191        AbstractMap.SimpleEntry<Integer, Integer> coord;
192        int count = 0;
193        for (Integer y = 1; y <= mida; y++) {
194            coord = new AbstractMap.SimpleEntry<>(x, y);
195            Integer numCjt = coordToNumCjt.get(coord); // Obtenir el id del conjunt de celes corresponent a les coordenades
196            ConjuntCella cjt = cjtCeles.get(numCjt); // Obtenir el conjunt de celes corresponent al id
197            Integer valCela = cjt.getValorCella(coord);
198            if (valCela == valor) {
199                count++;
200                if (count > 1) {
201                    return false;
202                }
203            }
204        }
205        return count == 1; 
206    }

508    public boolean finalitzar() {

510        
511        // primer, recorrer totes les columnes i files fent el comprovarFila i comprovarCol
512        for (Integer i = 1; i <= mida; i++) {

514            for (Integer x = 1; x < mida; x++) {
515                if (!comprovarFila(x, i)){
516                    return false;
517                }
518            }
519            for (Integer y = 1; y < mida; y++) {
520                if (!comprovarCol(y, i))
521                    return false;
522            }
523        }
524        // Despres anar a cada Cjt de celes i revisar que el resultat sigui el correcte
525        for (HashMap.Entry<Integer, ConjuntCella> entry : cjtCeles.entrySet()) {
526            ConjuntCella cjt = entry.getValue();
527            if (!cjt.comprobarRes())
528                return false;
529        }
530        // si tot es correcte, finalitzar la partida
531        return true;
532    }
