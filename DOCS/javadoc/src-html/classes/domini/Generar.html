011/**
012 * Representa un generador de Kenken.
013 * 
014 * A partir de les dades proporcionades, el generador de Kenken genera un Kenken amb les caracteristiques proposades.
015 * Algunes d'aquestes dades son el tamany del Kenken, el nombre maxim de celles que pot tenir un conjunt, el nombre de celes collocades i les operacions.
016 */
017public class Generar{
018    private static Generar instancia = null;
019    private Integer mida = 0;
020    private Integer nConjunts = 0;
021    private HashMap<AbstractMap.SimpleEntry<Integer, Integer>, Boolean> colocades;
022    private HashMap<AbstractMap.SimpleEntry<Integer, Integer>, Integer> solucio;
023    private HashMap<AbstractMap.SimpleEntry<Integer, Integer>, Integer> conjunts;
024
025    private HashMap<Integer, Integer> numCeles;
026    private HashMap<Integer, Integer> ops;
027    private HashMap<Integer, Integer> resultats;
028
029    private HashMap<Integer, Set<AbstractMap.SimpleEntry<Integer, Integer>>> cjt_celes;

052    /**
053     * Retorna un String que representa un Kenken generat a partir de les dades proporcionades.
054     * 
055     * @param mida Tamany del Kenken.
056     * @param maxConjunts Nombre maxim de celles que pot tenir un conjunt.
057     * @param ncColocades Nombre de celes collocades.
058     * @param operacions Operacions que poden tenir els conjunts.
059     * @return String
060     */
061    public String generarKenken(Integer mida, Integer maxConjunts, Integer ncColocades, Boolean[] operacions)
062    {
063        this.mida = mida;
064        colocades = new HashMap<AbstractMap.SimpleEntry<Integer, Integer>, Boolean>();
065        cjt_celes = new HashMap<Integer, Set<AbstractMap.SimpleEntry<Integer, Integer>>>();
066        conjunts = new HashMap<AbstractMap.SimpleEntry<Integer, Integer>, Integer>();
067        solucio = new HashMap<AbstractMap.SimpleEntry<Integer, Integer>, Integer>();
068        for(int i = 1; i <= mida; ++i){
069            for(int j = 1; j <= mida; ++j){
070                colocades.put(new AbstractMap.SimpleEntry<Integer, Integer>(i, j), false);
071                conjunts.put(new AbstractMap.SimpleEntry<Integer, Integer>(i, j), -1);
072                solucio.put(new AbstractMap.SimpleEntry<Integer, Integer>(i, j), -1);
073            }
074        }
075        this.omplirCeles(1, 1);
076        numCeles = new HashMap<Integer, Integer>();
077        this.generarConjunts(1, 1, maxConjunts, 0);
078        ops = new HashMap<Integer, Integer>();
079        this.generarOperacions(operacions, 0);
080        resultats = new HashMap<Integer, Integer>();
081        this.generarResultat(1, 1);
082        this.triarColocades(ncColocades);
083        return construirKenken();
084
085    }

087    /**
088     * Defineix quines celles estan collocades inicialment en el Kenken.
089     * 
090     * @param ncColocades Nombre de celes collocades que volem aconseguir.
091     * @return void
092     */
093    private void triarColocades(Integer ncColocades){
094        while(ncColocades != 0){
095            Integer min = 1; 
096            Integer max = mida; 
097
098            Random rand = new Random();
099            Integer firstRandomNumber =  rand.nextInt(max - min + 1) + min;
100            Integer x = firstRandomNumber;
101            Integer secondRandomNumber =  rand.nextInt(max - min + 1) + min;
102            Integer y = secondRandomNumber;
103            if(colocades.get(new AbstractMap.SimpleEntry<Integer, Integer>(x, y)) == false){
104                colocades.put(new AbstractMap.SimpleEntry<Integer, Integer>(x, y), true);
105                ncColocades = ncColocades - 1;
106            }
107        }
108    }

110    /**
111     * Omple recursivament, fent un backtracking, les celles del Kenken amb un valor aleatori satisfent les normes de no repetir el mateix valor per fila i columna.
112     * 
113     * @param x Coordenada x de la cella.
114     * @param y Coordenada y de la cella.
115     * @return Boolean
116     * 
117     */
118    private Boolean omplirCeles(Integer x, Integer y){
119        if(y > mida) return true;
120        else{
121            if(x > mida){
122                return omplirCeles(1, y+1);
123            } else {
124                int min = 1; 
125                int max = mida; 
126
127                Random rand = new Random();
128                int firstRandomNumber =  rand.nextInt(max - min + 1) + min;
129                int randomNumber =  firstRandomNumber;
130                boolean[] used = new boolean[max+1];
131                
132                for(int i = 0; i < mida; i++){
133                    while(!comprobarCol(x, randomNumber) || !comprobarFil(y, randomNumber) || used[randomNumber]){
134                        randomNumber = randomNumber + 1;
135                        if(randomNumber > max) randomNumber = min;
136                        if(randomNumber == firstRandomNumber) return false;
137                    }
138                    solucio.put(new AbstractMap.SimpleEntry<Integer, Integer>(x, y), randomNumber);
139                    used[randomNumber] = true;
140                    if(omplirCeles(x+1, y)) return true;
141                    solucio.put(new AbstractMap.SimpleEntry<Integer, Integer>(x, y), -1);
142                }
143                return false;
144            }
145        }
146    }

148    /**
149     * Comprova si un nombre n es pot collocar a la fila y.
150     * 
151     * @param y Fila on es vol collocar el nombre.
152     * @param n Nombre que es vol collocar.
153     * @return Boolean
154     */
155    private Boolean comprobarFil(Integer y, Integer n){
156        for(int i = 1; i <= mida; ++i){
157            if(solucio.get(new AbstractMap.SimpleEntry<Integer, Integer>(i, y)) == n) return false;
158        }
159        return true;
160    }
161
162    /**
163     * Comprova si un nombre n es pot collocar a la columna x.
164     * 
165     * @param x Columna on es vol collocar el nombre.
166     * @param n Nombre que es vol collocar.
167     * @return Boolean
168     */
169    private Boolean comprobarCol(Integer x, Integer n){
170        for(int i = 1; i <= mida; ++i){
171            if(solucio.get(new AbstractMap.SimpleEntry<Integer, Integer>(x, i)) == n) return false;
172        }
173        return true;
174    }

176    /**
177     * Genera els conjunts del Kenken recursivament tenint en compte no sobrepassar el maxim de celles per conjunt.
178     * 
179     * @param x Coordenada x de la cella.
180     * @param y Coordenada y de la cella.
181     * @param maxConjunts Nombre maxim de celles que pot tenir un conjunt.
182     * @param n Identificador del conjunt.
183     * @return void
184     */
185    private void generarConjunts(Integer x, Integer y, Integer maxConjunts, Integer n){
186        if(y > mida) {
187            nConjunts = n;
188            return;
189        }
190        else if(x > mida) generarConjunts(1, y+1, maxConjunts, n);
191        else if(conjunts.get(new AbstractMap.SimpleEntry<Integer, Integer>(x, y)) != -1){
192            if(x > mida && y > mida){
193                nConjunts = n;
194                return;
195            } else {
196                if(x > mida){
197                    generarConjunts(1, y+1, maxConjunts, n);
198                } else {
199                    generarConjunts(x+1, y, maxConjunts, n);
200                }
201            }
202        } else {
203            Integer min = 1; 
204            Integer max = maxConjunts; 
205
206            Random rand = new Random();
207            Integer firstRandomNumber =  rand.nextInt(max - min + 1) + min;
208
209            Integer randomNumber = dfs(firstRandomNumber, x, y, n, 0);
210            numCeles.put(n, randomNumber);
211            generarConjunts(x+1, y, maxConjunts, n+1);
212            
213        }
214        
215        
216    }

218    /**
219     * Afegeix una coordenada a un conjunt.
220     * 
221     * @param number Identificador del conjunt.
222     * @param coordinate Coordenada a afegir.
223     * @return void
224     */
225    private void addCoordinate(Integer number, AbstractMap.SimpleEntry<Integer, Integer> coordinate) {
226        // Get the set of coordinates associated with the number
227        Set<AbstractMap.SimpleEntry<Integer, Integer>> coordinates = cjt_celes.get(number);
228        if (coordinates == null) {
229            // If there's no set of coordinates associated with the number, create a new one
230            coordinates = new HashSet<>();
231        }
232        // Add the new coordinate to the set
233        coordinates.add(coordinate);
234        // Update the map with the modified set of coordinates
235        cjt_celes.put(number, coordinates);
236    }

240    /**
241     * Realitza un recorregut en profunditat per a generar els conjunts del Kenken.
242     * 
243     * @param numC Nombre de celles que ha de tenir el conjunt.
244     * @param x Coordenada x de la cella.
245     * @param y Coordenada y de la cella.
246     * @param n Identificador del conjunt.
247     * @param aux Nombre de celles que ja s'han afegit al conjunt.
248     * @return Integer
249     */
250    private Integer dfs(Integer numC, Integer x, Integer y, Integer n, Integer aux){
251        if(numC == aux) return aux;
252        else if(x < 1 || x > mida || y < 1 || y > mida || conjunts.get(new AbstractMap.SimpleEntry<Integer, Integer>(x, y)) != -1){
253            return aux;
254        } else {
255            conjunts.put(new AbstractMap.SimpleEntry<Integer, Integer>(x, y), n);
256            addCoordinate(n, new AbstractMap.SimpleEntry<Integer, Integer>(x, y));
257            aux = aux + 1;
258            if(aux == numC) return numC;
259            Integer min = 1; 
260            Integer max = 4; 
261
262            Random rand = new Random();
263            Integer firstRandomNumber =  rand.nextInt(max - min + 1) + min;
264
265            Integer first = dfs(numC, dirx(firstRandomNumber, x), diry(firstRandomNumber, y), n, aux);
266            if(first != numC){
267                Integer second = dfs(numC, dirx(firstRandomNumber+1, x), diry(firstRandomNumber+1, y), n, first);
268                if(second != numC){
269                    Integer third = dfs(numC, dirx(firstRandomNumber+2, x), diry(firstRandomNumber+2, y), n, second);
270                    if(third != numC){
271                        Integer fourth = dfs(numC, dirx(firstRandomNumber+3, x), diry(firstRandomNumber+3, y), n, third);
272                        if(fourth != numC){
273                            return fourth;
274                        } else return numC;
275                    } else return numC;
276                } else return numC;
277            } else return numC;
278             
279        }
280
281    }

283    /**
284     * Retorna la coordenada x de la cella seguent en funcio de la direccio.
285     * 
286     * @param dir Direccio de la cella.
287     * @param x Coordenada x de la cella.
288     * @return Integer
289     */
290    private Integer dirx(Integer dir, Integer x){
291        if(dir == 1) return x+1;
292        else if(dir == 2) return x-1;
293        else return x;
294    }

296    /**
297     * Retorna la coordenada y de la cella seguent en funcio de la direccio.
298     * 
299     * @param dir Direccio de la cella.
300     * @param y Coordenada y de la cella.
301     * @return Integer
302     */
303    private Integer diry(Integer dir, Integer y){
304        if(dir == 3) return y+1;
305        else if(dir == 4) return y-1;
306        else return y;
307    }

309    /**
310     * Genera les operacions dels conjunts del Kenken de forma recursiva, tenint en compte quines operacions podem utilitzar y el numero de celles dels conjunts.
311     * 
312     * @param operacions Operacions que poden tenir els conjunts.
313     * @param n Identificador del conjunt.
314     * @return void
315     */
316    private void generarOperacions(Boolean[] operacions, Integer n){
317        if(n == nConjunts) return;
318        else{
319            Integer min = 1;
320            Integer max = 6;
321
322            Random rand = new Random();
323            Integer randomNumber =  rand.nextInt(max - min + 1) + min;
324
325            Integer nC = numCeles.get(n);
326            switch (nC){
327                case 1:
328                    ops.put(n, 0);
329                    break;
330                case 2:
331                    while(!operacions[randomNumber]){
332                        randomNumber += 1;
333                        if(randomNumber > max) randomNumber = min;
334                    }
335                    ops.put(n, randomNumber);
336                    break;
337                default:
338                    if(randomNumber.equals(2) || randomNumber.equals(4)){
339                        randomNumber += 1;
340                    }
341                    while(!operacions[randomNumber]){
342                        randomNumber += 1;
343                        if(randomNumber.equals(2) || randomNumber.equals(4)){
344                            randomNumber += 1;
345                        }
346                        if(randomNumber > max) randomNumber = min;
347                    }
348                    ops.put(n, randomNumber);
349                    break;
350            }
351
352            generarOperacions(operacions, n+1);
353        }
354
355    }

357    /**
358     * Genera el resultat de les operacions dels conjunts del Kenken de forma recursiva, mirant la operacio del conjunt i els valors de les celles que formen part del conjunt.
359     * 
360     * @param x Coordenada x de la cella.
361     * @param y Coordenada y de la cella.
362     * @return void
363     */
364    private void generarResultat(Integer x, Integer y){
365        if(y > mida) return;
366        else{
367            if(x > mida){
368                generarResultat(1, y+1);
369            } else {
370                Integer res;
371                Integer idConj = conjunts.get(new AbstractMap.SimpleEntry<Integer, Integer>(x, y));
372                Integer num = solucio.get(new AbstractMap.SimpleEntry<Integer, Integer>(x, y));
373                Integer op = ops.get(idConj);
374                switch(op){
375                    case 0:
376                        resultats.put(idConj, num);
377                        break;
378                    case 1:
379                        res = resultats.get(idConj) != null ? resultats.get(idConj) : 0;
380                        resultats.put(idConj, res + num);
381                        break;
382                    case 2:
383                        res = resultats.get(idConj) != null ? resultats.get(idConj) : 0;
384                        if(res == 0){
385                            resultats.put(idConj, num);
386                        } else {
387                            Integer aux = Math.abs(res - num);
388                            resultats.put(idConj, aux);
389                        }
390                        break;
391                    case 3:
392                        res = resultats.get(idConj) != null ? resultats.get(idConj) : 1;
393                        resultats.put(idConj, res * num);
394                        break;
395                    case 4:
396                        res = resultats.get(idConj) != null ? resultats.get(idConj) : 0;
397                        if(res == 0){
398                            resultats.put(idConj, num);
399                        } else {
400                            if(num%res == 0) resultats.put(idConj, num/res);
401                            else if(res%num == 0) resultats.put(idConj, res/num);
402                            else{
403                                ops.put(idConj, 1);
404                                resultats.put(idConj, num+res);
405                            }
406                        }
407                        break;
408                    case 5:
409                        res = resultats.get(idConj) != null ? resultats.get(idConj) : 0;
410                        if(res == 0){
411                            resultats.put(idConj, num);
412                        } else {
413                            if(res < num) resultats.put(idConj, num);
414                        }
415                        break;
416                    case 6:
417                        res = resultats.get(idConj) != null ? resultats.get(idConj) : 0;
418                        if(res == 0){
419                            resultats.put(idConj, num);
420                        } else {
421                            if(res > num) resultats.put(idConj, num);
422                        }
423                        break;
424                }
425                generarResultat(x+1, y);
426            }
427        }
428    }

430    /**
431     * Construeix un String en format JSON que representa un Kenken.
432     * 
433     * @return String
434     */
435    private String construirKenken(){
436        JsonObject kenken = new JsonObject();
437        kenken.addProperty("N", mida);
438        kenken.addProperty("R", nConjunts);
439        JsonArray regions = new JsonArray();
440        for(int i = 0; i < nConjunts; ++i){
441            JsonObject regio = new JsonObject();
442            regio.addProperty("oper", ops.get(i));
443            regio.addProperty("result", resultats.get(i));
444            regio.addProperty("e", numCeles.get(i));
445            JsonArray coordenades = new JsonArray();
446            Set<AbstractMap.SimpleEntry<Integer, Integer>> cjt = cjt_celes.get(i);
447            for(AbstractMap.SimpleEntry<Integer, Integer> coord : cjt){
448                JsonObject crd = new JsonObject();
449                crd.addProperty("x", coord.getKey());
450                crd.addProperty("y", coord.getValue());
451                crd.addProperty("cont", solucio.get(coord));
452                if(colocades.get(coord)) crd.addProperty("ini", 1);
453                else crd.addProperty("ini", 0);
454                coordenades.add(crd);
455            }
456            regio.add("coordenades", coordenades);
457            regions.add(regio);
458        }
459        kenken.add("regions", regions);
460        return kenken.toString();
461    }
