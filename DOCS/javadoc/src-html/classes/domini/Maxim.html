001package classes.domini;
002
003import java.util.HashMap;
004import java.util.AbstractMap;
005
006/**
007 * Representa una operacio de maxim.
008 * 
009 * La divisio te un resultat i un nombre de solucions possibles.
010 * Tambe proporciona metodes per calcular el resultat del maxim i les possibilitats del resultat.
011 */
012public class Maxim extends Operacio {
013    private static Maxim instance;
014    private Integer solucions;
015
016    /**
017     * Constructor de la classe Maxim.
018     */
019    private Maxim () {
020    } 
021
022    /**
023     * Retorna la instancia del maxim.
024     * 
025     * @return Maxim
026     */
027    public static Maxim ObtenirInstancia() {
028        if (instance == null) {
029            instance = new Maxim();
030        }
031        return instance;
032    }
033
034    /**
035     * Calcula el resultat del maxim.
036     * 
037     * @param vector Vector amb els operands.
038     * @return double
039     */
040    public double Calcula(double[] vector) {
041        if(vector.length == 0) return -1;
042        double maxim = vector[0];
043        for (int i = 1; i < vector.length; i++) {
044            if (vector[i] > maxim) {
045                maxim = vector[i];
046            }
047        }
048        return maxim;
049    }
050    
051    /**
052     * Calcula les possibilitats del resultat del maxim.
053     * 
054     * @param resultat Resultat del maxim.
055     * @param solucions Nombre de solucions possibles.
056     * @param cjtCella Conjunt de celles a les que pertany l'operacio.
057     * @return Integer
058     */
059    public Integer calcularPossibilitats(Integer resultat, Integer solucions, ConjuntCella cjtCella){
060        Integer celesBuides = 0;
061        this.solucions = solucions;
062        Integer maxim = resultat;
063        int it = 0;
064        HashMap<AbstractMap.SimpleEntry<Integer, Integer>, Integer> valorsIni = new HashMap<AbstractMap.SimpleEntry<Integer, Integer>, Integer>();
065        AbstractMap.SimpleEntry<Integer, Integer> coordMaxim = null;
066        HashMap<Integer, AbstractMap.SimpleEntry<Integer, Integer>> array = new HashMap<Integer, AbstractMap.SimpleEntry<Integer, Integer>>();
067        for (AbstractMap.SimpleEntry<Integer, Integer> coord : cjtCella.getCoordenades()) {
068            Integer valor = cjtCella.getCella(coord).getSolucio();
069            array.put(it, coord);
070            if(valor != 0){
071                if(cjtCella.comprovarFilaColumnaPossibilitat(coord.getKey(), coord.getValue(), valor)) {
072                    valorsIni.put(coord, valor);
073                    if(valor == resultat) coordMaxim = coord;
074                }
075                else {
076                    return 0;
077                }
078            } else {
079                valorsIni.put(coord, 0);
080                celesBuides++;
081            }
082            ++it;
083        }
084
085        if(celesBuides == 0 && coordMaxim == null) return 0;
086        Integer aux = -1;
087        Integer possibilitats = calculRecursiu(celesBuides, valorsIni, maxim, coordMaxim, cjtCella, array, 0, aux);
088        
089        return possibilitats;
090    }
091 
092    /**
093     * Calcula les possibilitats del resultat del maxim de forma recursiva.
094     * 
095     * @param celesBuides Nombre de celles buides.
096     * @param valorsIni Valors inicials de les celles.
097     * @param maxim Resultat del maxim.
098     * @param coordMaxim Coordenades del maxim.
099     * @param cjtCella Conjunt de celles a les que pertany l'operacio.
100     * @param array Array amb les coordenades de les celles.
101     * @param it Iterador de l'array.
102     * @param aux Auxiliar per a la cerca del maxim.
103     * @return Integer
104     */
105    private Integer calculRecursiu(Integer celesBuides, HashMap<AbstractMap.SimpleEntry<Integer, Integer>, Integer> valorsIni, Integer maxim, AbstractMap.SimpleEntry<Integer, Integer> coordMaxim, ConjuntCella cjtCella, HashMap<Integer, AbstractMap.SimpleEntry<Integer, Integer>> array, Integer it, Integer aux){
106        if (celesBuides == 0 && coordMaxim != null) {
107            solucions = solucions + 1;
108            for(AbstractMap.SimpleEntry<Integer, Integer> coord : valorsIni.keySet()){
109                cjtCella.afegirPossibilitatTrue(coord, valorsIni.get(coord), solucions);
110            }
111            return 1;
112        } else if (celesBuides == 0 && coordMaxim == null) {
113            return 0;
114        } else {
115            
116            Integer possibilitats = 0;
117            if(valorsIni.get(array.get(it)) != 0) {
118                possibilitats += calculRecursiu(celesBuides, valorsIni, maxim, coordMaxim, cjtCella, array, it + 1, aux);
119            } else{
120                for(int i = maxim; i > 0; i--){
121                    if(cjtCella.setSolucio(array.get(it), i)){
122                        valorsIni.put(array.get(it), i);
123                        if(i == maxim && coordMaxim == null) {
124                            aux = it;
125                            coordMaxim = array.get(it);
126                        }
127                        if(cjtCella.comprovarFilaColumnaPossibilitat(array.get(it).getKey(), array.get(it).getValue(), i)) {
128                            possibilitats += calculRecursiu(celesBuides - 1, valorsIni, maxim, coordMaxim, cjtCella, array, it + 1, aux);
129                        }
130                        if(cjtCella.deleteSolucio(array.get(it))){
131                            valorsIni.put(array.get(it), 0);
132                            if(i == maxim && it == aux) {
133                                coordMaxim = null;
134                                aux = -1;
135                            }
136                        }
137                    }
138                }
139            }
140            return possibilitats;
141        }
142    }
